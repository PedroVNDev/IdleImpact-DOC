78  public static BigDouble Normalize(double mantissa, long exponent)
79  {
80  if (mantissa >= 1 && mantissa < 10 || !IsFinite(mantissa))
81  {
82  return new BigDouble(mantissa, exponent);
83  }
84 
85  if (IsZero(mantissa))
86  {
87  return Zero;
88  }
89 
90  var tempExponent = (long) Math.Floor(Math.Log10(Math.Abs(mantissa)));
91  //SAFETY: handle 5e-324, -5e-324 separately
92  if (tempExponent == DoubleExpMin)
93  {
94  mantissa = mantissa * 10 / 1e-323;
95  }
96  else
97  {
98  mantissa = mantissa / PowersOf10.Lookup(tempExponent);
99  }
100

306  public static BigDouble Add(BigDouble left, BigDouble right)
307  {
308  //figure out which is bigger, shrink the mantissa of the smaller by the difference in exponents, add mantissas, normalize and return
309 
310  //TODO: Optimizations and simplification may be possible, see https://github.com/Patashu/break_infinity.js/issues/8
311 
312  if (IsZero(left.Mantissa))
313  {
314  return right;
315  }
316 
317  if (IsZero(right.Mantissa))
318  {
319  return left;
320  }
321 
322  if (IsNaN(left) || IsNaN(right) || IsInfinity(left) || IsInfinity(right))
323  {
324  // Let Double handle these cases.
325  return left.Mantissa + right.Mantissa;
326  }
327 
328  BigDouble bigger, smaller;
329  if (left.Exponent >= right.Exponent)
330  {
331  bigger = left;
332  smaller = right;
333  }
334  else
335  {
336  bigger = right;
337  smaller = left;
338  }
339 
340  if (bigger.Exponent - smaller.Exponent > MaxSignificantDigits)
341  {
342  return bigger;
343  }
344 
345  //have to do this because adding numbers that were once integers but scaled down is imprecise.
346  //Example: 299 + 18
347  return Normalize(
348  Math.Round(1e14 * bigger.Mantissa + 1e14 * smaller.Mantissa *
349  PowersOf10.Lookup(smaller.Exponent - bigger.Exponent)),
350  bigger.Exponent - 14);
351  }

676  private static BigDouble PowInternal(BigDouble value, double other)
677  {
678  //UN-SAFETY: Accuracy not guaranteed beyond ~9~11 decimal places.
679 
680  //TODO: Fast track seems about neutral for performance. It might become faster if an integer pow is implemented, or it might not be worth doing (see https://github.com/Patashu/break_infinity.js/issues/4 )
681 
682  //Fast track: If (this.exponent*value) is an integer and mantissa^value fits in a Number, we can do a very fast method.
683  var temp = value.Exponent * other;
684  double newMantissa;
685  if (IsInteger(temp) && IsFinite(temp) && Math.Abs(temp) < ExpLimit)
686  {
687  newMantissa = Math.Pow(value.Mantissa, other);
688  if (IsFinite(newMantissa))
689  {
690  return Normalize(newMantissa, (long) temp);
691  }
692  }
693 
694  //Same speed and usually more accurate. (An arbitrary-precision version of this calculation is used in break_break_infinity.js, sacrificing performance for utter accuracy.)
695 
696  var newexponent = Math.Truncate(temp);
697  var residue = temp - newexponent;
698  newMantissa = Math.Pow(10, other * Math.Log10(value.Mantissa) + residue);
699  if (IsFinite(newMantissa))
700  {
701  return Normalize(newMantissa, (long) newexponent);
702  }
703 
704  //UN-SAFETY: This should return NaN when mantissa is negative and value is noninteger.
705  var result = Pow10(other * AbsLog10(value)); //this is 2x faster and gives same values AFAIK
706  if (Sign(value) == -1 && AreEqual(other % 2, 1))
707  {
708  return -result;
709  }
710 
711  return result;
712  }
